int nowl=1,nowr=0;
long long f[maxn+5],cost,Mid;
void move(int l,int r)
{
    while(nowl>l)
    {
        cost+=num[a[--nowl]];
        num[a[nowl]]++;
    }
    while(nowr<r)
    {
        cost+=num[a[++nowr]];
        num[a[nowr]]++;
    }
    while(nowl<l)
    {
        num[a[nowl]]--;
        cost-=num[a[nowl++]];
    }
    while(nowr>r)
    {
        num[a[nowr]]--;
        cost-=num[a[nowr--]];
    }
}
void dp(int l,int r,int L,int R)//ç”¨[l,r]çš„å†³ç­–ç‚¹æ›´æ–°[L,R]
{
    if(L>R)
        return;
    int mid=(L+R)>>1;
    int pos=0;
    long long minn=INF;
    for(int i=l;i<=r;i++)
    {
        move(i+1,mid);
        long long now=f[i]+cost+Mid;
        if(now<minn)
        {
            minn=now;
            pos=i;
        }
    }
    if(minn<f[mid])
    {
        f[mid]=minn;
        cnt[mid]=cnt[pos]+1;
    }
    dp(l,pos,L,mid-1);
    dp(pos,r,mid+1,R);
}
void CDQ(int l,int r)
{
    if(l>=r)
        return;
    int mid=(l+r)>>1;
    CDQ(l,mid);
    dp(l,mid,mid+1,r);
    CDQ(mid+1,r);
}

    scanf("%d%d",&n,&k);
    INF=1ll*n*(n-1)/2;

        Mid=(l+r)/2;

        CDQ(0,n);
        if(cnt[n]>k)
            l=Mid+1;
        else
            r=Mid-1;
    }
    Mid=l;

    f[0]=cnt[0]=0;
    CDQ(0,n);

    return 0;
}
