int q[maxn+5],ed[maxn+5],rt[maxn+5];
int cnt,num[maxn+5];
long long sum[maxn+5];
char s[maxn+5];
void insert(char *s,int now,int val)
{
    rt[now]=++tot;
    int u=tot,len=strlen(s+1);
    for(int i=1;i<=len;i++)
    {
        int c=s[i]-'a';
        if(!trie[u][c])
            trie[u][c]=++tot;
        u=trie[u][c];
    }
    ed[u]+=val;
}
void build(int root)
{
    int l=1,r=0;
    sum[root]=0;
    for(int i=0;i<26;i++)
    {
        if(trie[root][i])
        {
            ac[root][i]=trie[root][i];
            fail[ac[root][i]]=root;
            q[++r]=ac[root][i];
        }
        else
            ac[root][i]=root;
    }
    while(l<=r)
    {
        int u=q[l++];
        for(int i=0;i<26;i++)
        {
            if(trie[u][i])
            {
                ac[u][i]=trie[u][i];
                fail[ac[u][i]]=ac[fail[u]][i];
                q[++r]=ac[u][i];
            }
            else
                ac[u][i]=ac[fail[u]][i];
        }
        sum[u]=sum[fail[u]]+ed[u];
    }
}
int merge(int x,int y)
{
    if(!(x&&y))
        return x|y;
    ed[x]+=ed[y];
    for(int i=0;i<26;i++)

}
