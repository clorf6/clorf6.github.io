    return h[j]+sqrt(abs(i-j));
}
template<class T>
void swap(T &x,T &y)
{
    T temp=x;
    x=y;
    y=temp;
}
void solve(int l,int r,int L,int R)
{
    if(l>r)
        return ;
    int mid=(l+r)>>1,pos=0;
    double las=0;
    for(int i=L;i<=mid&&i<=R;i++)
    {
        double now=calc(mid,i);
        if(las<now)
        {
            las=now;
            pos=i;
        }
    }
    ans[mid]=max(ans[mid],las);
    solve(l,mid-1,L,pos);
    solve(mid+1,r,pos,R);
}
int main()
{
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
        scanf("%d",&h[i]);
    solve(1,n,1,n);
    for(int i=1;i<=n/2;i++)
    {
        swap(h[i],h[n-i+1]);
        swap(ans[i],ans[n-i+1]);
    }
    solve(1,n,1,n);
    for(int i=n;i>=1;i--)
        printf("%d\n",(int)ceil(ans[i]-h[i]));
    return 0;
}

{

    int z;
    int col;
}s[maxn+5];
bool cmp(edge a,edge b)
{
    if(a.z==b.z)
        return a.col<b.col;
    return a.z<b.z;
}
int n,m,need,maxx,k;
int fa[maxn+5],ans,sum;
int getfa(int x)
{
    if(fa[x]==x)
        return x;
    return fa[x]=getfa(fa[x]);
}
bool check(int x)
{
    for(int i=1;i<=m;i++)
        if(!s[i].col)
            s[i].z+=x;
    for(int i=1;i<=n+1;i++)
        fa[i]=i;
    sum=k=0;
    sort(s+1,s+m+1,cmp);
    int tot=0;
    for(int i=1;i<=m;i++)
    {
        if(tot==n-1)
            break;
        int a=getfa(s[i].x);
        int b=getfa(s[i].y);
        if(a==b)
            continue;
        else
        {
            tot++;
            fa[a]=b;
            sum+=s[i].z;
            k+=(s[i].col^1);
        }
    }
    return k>=need;
}
void clean(int x)
{
    for(int i=1;i<=m;i++)
        if(!s[i].col)

}
